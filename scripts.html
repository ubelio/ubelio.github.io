EnableBitlockerSystemDriveC.ps1

This script checks BitLocker status on the system drive (C:), enables encryption if needed, detects TPM availability, and backs up the recovery key to Active Directory.

# Check BitLocker status
$bitlockerStatus = Get-BitLockerVolume -MountPoint "C:"
if ($bitlockerStatus.ProtectionStatus -eq 'Off') {
    # Enable BitLocker with TPM and AD key backup
    Enable-BitLocker -MountPoint "C:" -EncryptionMethod XtsAes256 -TpmProtector
    Backup-BitLockerKeyProtector -MountPoint "C:"
}

EnableBitlockerDataDriveD.ps1

This script detects and encrypts data drives (D:, E:, F:) and removable drives using the same logic as the system drive.

# Encrypt data and removable drives
$drives = Get-BitLockerVolume | Where-Object { $_.MountPoint -match "D:|E:|F:" }
foreach ($drive in $drives) {
    if ($drive.ProtectionStatus -eq 'Off') {
        Enable-BitLocker -MountPoint $drive.MountPoint -EncryptionMethod XtsAes256 -PasswordProtector
        Backup-BitLockerKeyProtector -MountPoint $drive.MountPoint
    }
}
